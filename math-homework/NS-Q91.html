from itertools import permutations
l = [m for i,j,k in permutations(range(5),3) if (m:=i*100+j*10+k) > 100]

print(f'(1):{len(l)}')
print(
    '(2):'
    f'<奇数:{len([n for n in l if n % 2])}>,'
    f'<偶数:{len([n for n in l if not n % 2])}>'
    )
print(
    '(3):'
    f'<9の倍数:{len([n for n in l if not n % 9])}>,'
    f'<4の倍数:{len([n for n in l if not n % 4])}>'
    )
print(f'(4):{l.index(213)+1}')
